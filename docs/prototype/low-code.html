集成示例

以下是在不同低代码平台中集成API的代码示例

// 使用fetch API调用月度销售报表
async function getMonthlySalesReport(params) {
  const response = await fetch('https://api.datainsight.com/api/v1/reports/monthly-sales', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': 'Bearer YOUR_API_KEY'
    },
    body: JSON.stringify(params)
  });

  if (!response.ok) {
    throw new Error('API request failed');
  }

  return await response.json();
}

// 示例用法
getMonthlySalesReport({
  startDate: '2023-01-01',
  endDate: '2023-12-31',
  groupBy: 'month'
})
.then(data => {
  console.log('Monthly sales report:', data);
  // 在这里处理数据，例如更新图表或表格
})
.catch(error => {
  console.error('Error fetching monthly sales report:', error);
});
