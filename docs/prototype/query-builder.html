SQL预览

SELECT
    o.order_id,
    o.customer_id,
    o.order_date,
    c.customer_name
FROM
    orders o
INNER JOIN
    customers c ON o.customer_id = c.customer_id
WHERE
    o.order_date >= '2023-01-01'
    AND o.status = '已完成'
ORDER BY
    o.order_date DESC
LIMIT 100 OFFSET 0;
